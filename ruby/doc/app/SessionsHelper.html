def admin_user
  unless current_user.admin?
    respond_to do |format|
      format.html { redirect_to root_path, notice: "Please sign in." }
      format.json { render json: {:status => false, :message => ["You are not an Admin"], :userId=>0}, status: :unauthorized }
    end
  end
end
